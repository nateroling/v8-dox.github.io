38  inline int GetInt() {
39  // This way of variable-length encoding integers does not suffer from branch
40  // mispredictions.
41  uint32_t answer = GetUnalignedInt();
42  int bytes = answer & 3;
43  Advance(bytes);
44  uint32_t mask = 0xffffffffu;
45  mask >>= 32 - (bytes << 3);
46  answer &= mask;
47  answer >>= 2;
48  return answer;
49  }
