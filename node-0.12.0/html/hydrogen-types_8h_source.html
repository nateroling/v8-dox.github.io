44  // Return the weakest (least precise) common type.
45  HType Combine(HType other) const V8_WARN_UNUSED_RESULT {
46  return HType(static_cast<Kind>(kind_ & other.kind_));
47  }
48 
49  bool Equals(HType other) const V8_WARN_UNUSED_RESULT {
50  return kind_ == other.kind_;
51  }
52 
53  bool IsSubtypeOf(HType other) const V8_WARN_UNUSED_RESULT {
54  return Combine(other).Equals(other);
55  }
