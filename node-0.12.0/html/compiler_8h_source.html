398  // Flags using template class BitField<type, start, length>. All are
399  // false by default.
400  //
401  // Compilation is either eager or lazy.
402  class IsLazy: public BitField<bool, 0, 1> {};

413  class SupportsDeoptimization: public BitField<bool, 6, 1> {};

415  // initial mode setting.
416  class IsCompilingForDebugging: public BitField<bool, 7, 1> {};

418  class IsCalling: public BitField<bool, 8, 1> {};
419  // If the compiled code contains calls that require building a frame
420  class IsDeferredCalling: public BitField<bool, 9, 1> {};
421  // If the compiled code contains calls that require building a frame
