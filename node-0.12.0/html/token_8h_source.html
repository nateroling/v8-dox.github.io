13 // TOKEN_LIST takes a list of 3 macros M, all of which satisfy the
14 // same signature M(name, string, precedence), where name is the
15 // symbolic token name, string is the corresponding syntactic symbol
16 // (or NULL, for literals), and precedence is the precedence (or 0).
17 // The parameters are invoked for token categories as follows:
18 //
19 // T: Non-keyword tokens
20 // K: Keyword tokens
21 
22 // IGNORE_TOKEN is a convenience macro that can be supplied as
23 // an argument (at any position) for a TOKEN_LIST call. It does
24 // nothing with tokens belonging to the respective category.
25 
26 #define IGNORE_TOKEN(name, string, precedence)

164 class Token {
165  public:
166  // All token values.
167 #define T(name, string, precedence) name,
168  enum Value {
169  TOKEN_LIST(T, T)
170  NUM_TOKENS
171  };
172 #undef T
173 
174  // Returns a string corresponding to the C++ token name
175  // (e.g. "LT" for the token LT).
176  static const char* Name(Value tok) {
177  DCHECK(tok < NUM_TOKENS); // tok is unsigned
178  return name_[tok];
179  }
180 
181  // Predicates
182  static bool IsKeyword(Value tok) {
183  return token_type[tok] == 'K';
184  }
185 
186  static bool IsAssignmentOp(Value tok) {
187  return INIT_VAR <= tok && tok <= ASSIGN_MOD;
188  }
189 
190  static bool IsBinaryOp(Value op) {
191  return COMMA <= op && op <= MOD;
192  }
193 
194  static bool IsTruncatingBinaryOp(Value op) {
195  return BIT_OR <= op && op <= ROR;
196  }
197 
198  static bool IsCompareOp(Value op) {
199  return EQ <= op && op <= IN;
200  }
201 
202  static bool IsOrderedRelationalCompareOp(Value op) {
203  return op == LT || op == LTE || op == GT || op == GTE;
204  }
205 
206  static bool IsEqualityOp(Value op) {
207  return op == EQ || op == EQ_STRICT;
208  }
209 
210  static bool IsInequalityOp(Value op) {
211  return op == NE || op == NE_STRICT;
212  }
213 
214  static bool IsArithmeticCompareOp(Value op) {
215  return IsOrderedRelationalCompareOp(op) ||
216  IsEqualityOp(op) || IsInequalityOp(op);
217  }
218 
219  static Value NegateCompareOp(Value op) {
220  DCHECK(IsArithmeticCompareOp(op));
221  switch (op) {
222  case EQ: return NE;
223  case NE: return EQ;
224  case EQ_STRICT: return NE_STRICT;
225  case NE_STRICT: return EQ_STRICT;
226  case LT: return GTE;
227  case GT: return LTE;
228  case LTE: return GT;
229  case GTE: return LT;
230  default:
231  UNREACHABLE();
232  return op;
233  }
234  }
235 
236  static Value ReverseCompareOp(Value op) {
237  DCHECK(IsArithmeticCompareOp(op));
238  switch (op) {
239  case EQ: return EQ;
240  case NE: return NE;
241  case EQ_STRICT: return EQ_STRICT;
242  case NE_STRICT: return NE_STRICT;
243  case LT: return GT;
244  case GT: return LT;
245  case LTE: return GTE;
246  case GTE: return LTE;
247  default:
248  UNREACHABLE();
249  return op;
250  }
251  }
252 
253  static bool IsBitOp(Value op) {
254  return (BIT_OR <= op && op <= SHR) || op == BIT_NOT;
255  }
256 
257  static bool IsUnaryOp(Value op) {
258  return (NOT <= op && op <= VOID) || op == ADD || op == SUB;
259  }
260 
261  static bool IsCountOp(Value op) {
262  return op == INC || op == DEC;
263  }
264 
265  static bool IsShiftOp(Value op) {
266  return (SHL <= op) && (op <= SHR);
267  }
268 
269  // Returns a string corresponding to the JS token string
270  // (.e., "<" for the token LT) or NULL if the token doesn't
271  // have a (unique) string (e.g. an IDENTIFIER).
272  static const char* String(Value tok) {
273  DCHECK(tok < NUM_TOKENS); // tok is unsigned.
274  return string_[tok];
275  }
276 
277  // Returns the precedence > 0 for binary and compare
278  // operators; returns 0 otherwise.
279  static int Precedence(Value tok) {
280  DCHECK(tok < NUM_TOKENS); // tok is unsigned.
281  return precedence_[tok];
282  }
283 
284  private:
285  static const char* const name_[NUM_TOKENS];
286  static const char* const string_[NUM_TOKENS];
287  static const int8_t precedence_[NUM_TOKENS];
288  static const char token_type[NUM_TOKENS];
289 };
